STC8系列

外设介绍

STC8系列单片机有4个全双工异步通信接口（串口1\2\3\4）。每个串口的通讯口均可以通过功能管脚的切换功能切换到多组端口，从而实现一个串口复用为多个串口的功能。部分型号可能有所不同。

串口1，工作模式有四种：同步移位串行模式、可变波特率8位数据模式、固定波特率9位数据模式、可变波特率9位数据模式。有四级中断优先级（0~3），在开启中断后，需要软件清理标志位。由定时器1、2做波特率发生器。可以进行多机通信，中继广播。

定时器/计数器1，工作模式有两种：可变波特率8位数据模式、固定波特率9位数据模式。有四级中断优先级（0~3），开启中断后，需要软件清理标志位。由定时器2做波特率发生器。可中继广播。STC8G、STC8H系列还支持时钟8位预分频功能。

定时器/计数器2，工作模式有两种：可变波特率8位数据模式、固定波特率9位数据模式。有四级中断优先级（0~3），开启中断后，需要软件清理标志位。由定时器2做波特率发生器。可中继广播。STC8G、STC8H系列还支持时钟8位预分频功能。

定时器/计数器3，工作模式有两种：可变波特率8位数据模式、固定波特率9位数据模式。有四级中断优先级（0~3），开启中断后，需要软件清理标志位。由定时器2做波特率发生器。可中继广播。STC8G、STC8H系列还支持时钟8位预分频功能。

定时器/计数器4，工作模式有两种：可变波特率8位数据模式、固定波特率9位数据模式。有四级中断优先级（0~3），开启中断后，需要软件清理标志位。由定时器2做波特率发生器。可中继广播。STC8G、STC8H系列还支持时钟8位预分频功能。

函数名 描述

UART_InitType 结构体说明
typedef struct 
{
  UARTMode_Type Mode;        //工作模式
  UARTBRTGen_Type BRTGen;    //波特率发生器
  UARTBRTMode_Type BRTMode;  //波特率发生器时钟模式
  FUNSTATE BRTDouble;        //波特率加倍（只有串口1有）
  uint32 BaudRate;           //波特率值
  FUNSTATE MulitComm; 		 //多机通信（只有串口1有）
  FUNSTATE Relay; 		     //中继广播
  FUNSTATE RxEnable;         //接收使能控制位
} UART_InitType;
UARTMode_Type 枚举体说明
typedef enum
{
	UART_SyncShift  = 0x00,  //模式0：同步移位串行模式
	UART_8bit_BRTx  = 0x01,	 //模式1：8位波特率可变模式
	UART_9bit_Fixed = 0x02,  //模式2：9位波特率固定模式
	UART_9bit_BRTx  = 0x03   //模式3：9位波特率可变模式
}	UARTMode_Type;
UARTBRTGen_Type 枚举体说明
typedef enum
{
	UART_BRT_TIM1 = 0x00,   //定时器1做波特率发生器
	UART_BRT_TIM2 = 0x01,   //定时器2做波特率发生器
	UART_BRT_TIM3 = 0x02,	//定时器3做波特率发生器
	UART_BRT_TIM4 = 0x03 	//定时器4做波特率发生器
}	UARTBRTGen_Type;
UARTBRTMode_Type 枚举体说明
typedef enum
{
    UART_BRT_1T  = 0x00,   // 波特率发生器1T模式 
    UART_BRT_12T = 0x01    // 波特率发生器12T模式
}	UARTBRTMode_Type;

1. 串口初始化演示

void STC8x_UART_Config(void)
{
	UART_InitType UART_InitStruct={0};	// 定义串口初始化结构体

	/* UART1 TXD */
	GPIO_MODE_OUT_PP(GPIO_P3,Pin_1); //P31口设置为推挽输出
	/* UART1 RXD */
	GPIO_MODEIN_FLOATING(GPIO_P3,Pin_0); //P30口设置为浮空输入
    
	UART_InitStruct.Mode = UART_8bit_BRTx;  // 工作模式： 8位波特率可变模式
	UART_InitStruct.BRTGen = UART_BRT_TIM1; // 波特率发生器：定时器1
	UART_InitStruct.BRTMode = UART_BRT_1T;  // 波特率发生器模式：1T模式
	UART_InitStruct.BaudRate = 115200;  // 波特率值：115200
	UART_InitStruct.RxEnable = ENABLE;  // 接收使能控制位：开启接收
	UART1_Init(&UART_InitStruct);   //串口1初始化函数，传入结构体地址
	NVIC_UART1_Init(NVIC_PR0,ENABLE); //串口1中断初始化函数，0优先级，开启中断。
    
}

中断法(建议使用)

uint8_t Re dat = 0; // 定义一个无符号8位的变量

void UART1_ISRQ_Handler(void) // 串口1中断服务函数
{
  UART1_CLEAR_BUSY_FLAG();  // 串口1清理忙标志
  if (UART1_GET_TX_FLAG())  // 判断串口1输出标志位
  {
    UART1_CLEAR_TX_FLAG();  // 串口1清理输出标志位
  }
  if (UART1_GET_RX_FLAG())  // 判断串口1接收标志位
  {
    UART1_CLEAR_RX_FLAG();  // 串口1清理接收标志位
    UART1_Rcv_Byte(&Re dat); // 读取接收的字节，并存入变量Re dat中
  }
}

查询法(不建议使用)

#include "8051_ELL_Lib.h"

uint8_t Redat = 0; // 定义一个无符号8位的变量

int main(void)  //main 函数
{
	STC8Ax_PER_Init(); // 调用MCU外设初始化函数
	
	for(;;)  //无限循环体
	{
		UART1_CLEAR_BUSY_FLAG();  // 串口1清理忙标志
        while(!UART1_GET_RX_FLAG());
        UART1_CLEAR_RX_FLAG();  // 串口1清理接收标志位
   		UART1_Rcv_Byte(&Redat); // 读取接收的字节，并存入变量Redat中
	}
}

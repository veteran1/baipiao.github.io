8051ELL库-组件篇-按键扫描

8051MCU
8051 ELL函数库指南
发布日期:   2021-03-14

设备驱动介绍

按键，是我们日常开发中最常用的硬件设备之一，所以驱动按键成为了我们最平常也是最重要的一环。本驱动旨在帮助开发者快速实现按键驱动，简洁灵活，可拓展。

详细内容

目前本驱动支持短按触发功能，可以配置高低电平触发，按下后立即生效或放手以后才生效，可配合TMT组件使用，非常灵活。用户只需要按照给定的类型注册按键和按键事件函数，并做好初始化工作，即可实现按键功能！

实现方法，同样是利用了时间片扫描来消抖，所以需要占用一个定时器中断，或者在主函数中隔一段时间运行一次。


API手册

1. 按键扫描初始化函数（注册按键）

函数名 描述
FSCSTATE KeyScan_Init(KeyScan_Type * keyScan) 按键扫描初始化函数

KeyScan_Type 结构体说明

typedef struct
{
	uint8_t        GPIO;  /* GPIO IO口组号 GPIO_P0~GPIO_P07*/
	uint8_t        Pin;   /* IO引脚号 Pin0~Pin07 */
	Sem_Effect      EffMode; /* 按键事件的生效模式 */
	Sem_Tri         TriMode; /* 按键扫描的触发模式 */
	KeyHandle_Func  HandleFunc;  /* 按键的事件函数 */
}KeyScan_Type;

Sem_Effect 枚举体说明

typedef enum 
{
  Effect_Press   = 0x00, /* 按下后立刻生效 */
  Effect_Release = 0x01  /* 松手后立刻生效 */
} Sem_Effect;

KeyScan_Type 结构体说明

typedef enum 
{
  Tri_High_level = 0x00, /* 高电平触发 */
  Tri_Low_level  = 0x01  /* 低电平触发 */
} Sem_Tri;

2. 按键扫描操作函数

函数名 描述
FSCSTATE KeyScan.Run() 按键扫描运行函数
FSCSTATE KeyScan.ReadPin_P0() 按键扫描读取P0口值函数
FSCSTATE KeyScan.ReadPin_P1() 按键扫描读取P1口值函数
FSCSTATE KeyScan.ReadPin_P2() 按键扫描读取P2口值函数
FSCSTATE KeyScan.ReadPin_P3() 按键扫描读取P3口值函数
FSCSTATE KeyScan.ReadPin_P4() 按键扫描读取P4口值函数
FSCSTATE KeyScan.ReadPin_P5() 按键扫描读取P5口值函数
FSCSTATE KeyScan.ReadPin_P6() 按键扫描读取P6口值函数
FSCSTATE KeyScan.ReadPin_P7() 按键扫描读取P7口值函数

代码演示

  1. 方式一，利用定时器中断，来实现按键扫描的去抖操作，硬件上由P54连接按键，高电平触发按键

在Lib_CFG文件里，配置相关参数 ；


/*--------------------------------------------------------
| @Description: STC8x Driver config                      |
--------------------------------------------------------*/

/**
  * @name    KeyScan
  * @brief   You need to config key scan.  
***/
#define  DRV_LIB_KEYSCAN_CTRL        (1)    // "1" 开启, "0"关闭 
#define  DRV_LIB_KEYSCAN_NUM_MAX     (8)    // 按键的最大数量



static void STC8x_TIMER_Config(void) /* 初始化定时器0 20ms终端一次 */
{
	TIMER_InitType TIMER_InitStruct={0};
	
	TIMER_InitStruct.Mode = TIMER_16BitAutoReload;
	TIMER_InitStruct.Value = 20000;     //20ms 系统时钟频率设置24M
	TIMER_InitStruct.Run = ENABLE;
	TIMER0_Init(&TIMER_InitStruct);
	NVIC_TIMER0_Init(NVIC_PR0,ENABLE);

}

void TIMER0_ISRQ_Handler(void) /* 定时器0中断函数 */
{
	KeyScan.ReadPin_P5();  /* 由于我们是P54连的按键，就读取P5的IO值 */
}

void Key_Handle(void) /* 按键事件函数 */
{
	GPIO_TOGGLE_PIN(GPIO_P5,Pin_5);  /* 翻转P55的IO电平 */
	UART1_Send_Byte(64);  /* 串口发送64的数值 */
}

int main(void)
{
	KeyScan_Type KeyScan_P54 = {0}; /* 定义一个按键结构体 */

	STC8x_TIMER_Config(); /* 调用定时器0初始化函数 */
    NVIC_GLOBAL_ENABLE(); /* 初始化总中断 */

    GPIO_MODE_IN_FLOATING(GPIO_P5,Pin_4);  //P54 浮空输入
	GPIO_MODE_OUT_PP(GPIO_P5,Pin_5);  //P55  推挽输出
	
	KeyScan_P54.GPIO = GPIO_P5;  /* 指明P5 */
	KeyScan_P54.Pin = Pin_4;  /* 指明Pin4 */
	KeyScan_P54.EffMode = Effect_Release;  /* 选择松手生效 */
	KeyScan_P54.TriMode = Tri_High_level;  /* 选择高电平触发 */
	KeyScan_P54.HandleFunc = Key_Handle; /* 指明按键事件函数 */
	
	KeyScan_Init(&KeyScan_P54);  /* 注册按键 */
	
    for(;;)
    {
		KeyScan.Run();

    }

}
  1. 方式二，在主函数中每隔一段时间扫描一次（我们采用TMT组件），来实现按键扫描的去抖操作，硬件上由P54连接按键，高电平触发按键

在Lib_CFG文件里，配置相关参数 ；


/*--------------------------------------------------------
| @Description: STC8x Driver config                      |
--------------------------------------------------------*/

/**
  * @name    KeyScan
  * @brief   You need to config key scan.  
***/
#define  DRV_LIB_KEYSCAN_CTRL        (1)    // "1" 开启, "0"关闭 
#define  DRV_LIB_KEYSCAN_NUM_MAX     (8)    // 按键的最大数量

/*--------------------------------------------------------
| @Description: STC8x Componets config                   |
--------------------------------------------------------*/

/**
  * @name    TMT
  * @brief   Task time tool
***/
#define  COM_LIB_TMT_CTRL              (1)    // "1" 开启, "0"关闭 
#define  COM_LIB_TMT_TASK_NUM_MAX      (8)    // 最大任务数量
#define  COM_LIB_NVIC_TIMER_ISR_ENABLE    ET0 = 1  //定时器中断打开（根据你使用的定时器来选择）
#define  COM_LIB_NVIC_TIMER_ISR_DISABLE   ET0 = 0  //定时器中断关闭（根据你使用的定时器来选择）



static void STC8x_TIMER_Config(void) /* 初始化定时器0 20ms终端一次 */
{
	TIMER_InitType TIMER_InitStruct={0};
	
	TIMER_InitStruct.Mode = TIMER_16BitAutoReload;
	TIMER_InitStruct.Value = 1000;     //1ms 系统时钟频率设置24M
	TIMER_InitStruct.Run = ENABLE;
	TIMER0_Init(&TIMER_InitStruct);
	NVIC_TIMER0_Init(NVIC_PR0,ENABLE);

}

void TIMER0_ISRQ_Handler(void) /* 定时器0中断函数 */
{
    /*
    调用TMT节拍函数，为TMT提供心跳节拍，
    节拍时间根据你设置的定时器中断时间来决定
    */
    TMT.Tick(); 

}

void Key_Handle(void) /* 按键事件函数 */
{
	GPIO_TOGGLE_PIN(GPIO_P5,Pin_5);  /* 翻转P55的IO电平 */
	UART1_Send_Byte(64);  /* 串口发送64的数值 */
}

int main(void)
{
	KeyScan_Type KeyScan_P54 = {0}; /* 定义一个按键结构体 */

	STC8x_TIMER_Config(); /* 调用定时器0初始化函数 */
    NVIC_GLOBAL_ENABLE(); /* 初始化总中断 */
    TMT_Init();  /* 初始化TMT组件 */

    GPIO_MODE_IN_FLOATING(GPIO_P5,Pin_4);  //P54 浮空输入
	GPIO_MODE_OUT_PP(GPIO_P5,Pin_5);  //P55  推挽输出
	
	KeyScan_P54.GPIO = GPIO_P5;  /* 指明P5 */
	KeyScan_P54.Pin = Pin_4;  /* 指明Pin4 */
	KeyScan_P54.EffMode = Effect_Release;  /* 选择松手生效 */
	KeyScan_P54.TriMode = Tri_High_level;  /* 选择高电平触发 */
	KeyScan_P54.HandleFunc = Key_Handle; /* 指明按键事件函数 */
	KeyScan_Init(&KeyScan_P54);  /* 注册按键 */
	
    TMT.Create(KeyScan.ReadPin_P5,20); /* 20ticks 运行一次 */

    for(;;)
    {
		KeyScan.Run();
        TMT.Run();
    }

}
